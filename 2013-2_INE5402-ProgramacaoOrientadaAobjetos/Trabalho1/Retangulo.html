12 public class Retangulo 
13 {
14         private double base, altura; // atributos
15         
16         public Retangulo()
17         {
18                 base = 0.0;
19                 altura = 0.0;
20         }
21         
22         public Retangulo( double base, double altura )
23         {
24                 this.base = base;
25                 this.altura = altura;
26         }
27         
28         public void altereBase( double base )
29         {
30                 this.base = base;
31         }
32         
33         public void altereAltura( double altura )
34         {
35                 this.altura = altura;
36         }
37         
38         public double informeBase()
39         {
40                 return base;
41         }
42         
43         public double informeAltura()
44         {
45                 return altura;
46         }
47         
48         public double forneceArea()
49         {
50                 return base * altura;
51         }
52         
53         public double fornecePerimetro()
54         {
55                 return 2 * base + 2 * altura;
56         }
57         
58 } // end class Retangulo
